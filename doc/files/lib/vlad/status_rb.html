  def status_cmd(svr)

    master = "master"
    worker = "worker"
    psub = "**REPLACE**"
    
    ps = []
    ps << %(ps aux)
    ps << %(grep "#{svr}")
    ps << %(grep "#{psub}")
    ps << %(grep -v grep)
    ps << %(wc -l)

    cmd = []
    cmd << %(echo "Getting status for #{svr} on `hostname`")
    cmd << %(echo " #{master} processes running: `#{ps.join(" | ").gsub(psub, master)}`")
    cmd << %(echo " #{worker} processes running: `#{ps.join(" | ").gsub(psub, worker)}`")
    cmd.join(" && ")

  end
